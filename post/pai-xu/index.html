冒泡排序

//冒泡排序
void bubbleSort(vector<int> &nums) {
    int n = nums.size();

    for(int i = 0; i < n; i++) { //n个数字，比较N次，每次将一个元素放到正确的位置
        int flag = false;
        for(int j = 0; j < n - i - 1; j++) {
            if(nums[j] > nums[j + 1]){
                swap(nums[j],nums[j + 1]);
                flag = true;
            }
        }
        if(!flag) break;

    }
}

插入排序

//插入排序
void insertionSort(vector<int> &nums) {
    int n = nums.size();

    for (int i = 1; i < n; i++)
    {
        int temp = nums[i];
        int j = i; //寻找插入位置

        while (j - 1 >= 0 && temp < nums[j - 1])
        {
            nums[j] = nums[j - 1];
            j--;
        }
       
        nums[j] = temp;
    }
}

选择排序

//选择排序
void selectionSort(vector<int> &nums) {
    int n = nums.size();

    for (int i = 0; i < n - 1; i++) //因为在底下是从i 的后面找，所以这里i < n - 1
    {
        int min = i;

        for (int j = i + 1; j < n; j++) //
        {
            if (nums[j] < nums[min])
            {
                min = j;
            }
            
        }
        swap(nums[i],nums[min]); 
    }
}

希尔排序

//希尔排序
void shellSort(vector<int> &nums) {
    int n = nums.size();

    for (int gap = n / 2; gap > 0; gap /= 2)
    {
        for (int i = gap; i < n; i++)
        {
            int j = i; // 寻找插入位置
            int temp = nums[j];

            while (j - gap >= 0 && nums[j - gap] > temp)
            {
                    //swap(nums[j], nums[j - gap]);
                nums[j] = nums[j - gap];
                j -= gap;
            }
            nums[j] = temp;
        }
    }
}

归并排序

归并排序分为两个步骤，先分成左右两边，再合并起来。

void mergeSort(vector<int> &nums) {
    vector<int> temp(nums.size(), 0);
    mergeSortSort(nums, 0, nums.size() - 1, temp);
}

void mergeSortSort(vector<int> &nums,int left, int right, vector<int> &temp) {
    if (left < right) {
        cout<<"left: "<<left<<"  "<<"right: "<<right<<endl;
        cout<<"left: "<<left<<"  "<<"right: "<<right<<endl;
        int mid = (left + right) / 2;
        //分开
        mergeSortSort(nums, left, mid, temp);
        mergeSortSort(nums, mid + 1, right, temp);
        //合并
        mergeSortMerge(nums, left, mid, right, temp);
    }
}

void mergeSortMerge(vector<int> &nums,int left, int mid, int right, vector<int> &temp) {
    int i = left;
    int j = mid + 1;
    int t = 0;

    //将两个vector里面较小的拿出来放进temp里
    while (i <= mid && j <= right)
    {
        if (nums[i] <= nums[j])
        {
            temp[t++] = nums[i++];
        }
        else {
            temp[t++] = nums[j++];
        }
        
    }

    while (i <= mid) 
    {
        temp[t++] = nums[i++];
    }

    while (j <= right)
    {
        temp[t++] = nums[j++];
    }

    t = 0;
    while (left <= right)
    {
        nums[left++] = temp[t++];
    }
}

快速排序

//快速排序
void quickSortSort(vector<int> &nums, int left,int right);
void quickSortSort2(vector<int> &nums, int start, int end);

void quickSort(vector<int> &nums) {
    quickSortSort(nums, 0, nums.size() - 1);
}

//https://wiki.jikexueyuan.com/project/easy-learn-algorithm/fast-sort.html
void quickSortSort(vector<int> &nums, int left,int right)
{
    if(left > right)
       return;

    int temp = nums[left]; //temp中存的就是基准数
    int i = left;
    int j = right;

    while(i != j)
    {
       //顺序很重要，要先从右边开始找
        while(nums[j] >= temp && i < j)
            j--;
       //再找右边的
        while(nums[i] <= temp && i < j)
            i++;
       //交换两个数在数组中的位置
       if(i < j)
       {
           swap(nums[i], nums[j]);
       }
    }
    //最终将基准数归位
    nums[left] = nums[i];
    nums[i] = temp;

    quickSortSort(nums, left, i-1);//继续处理左边的，这里是一个递归的过程
    quickSortSort(nums, i+1, right);//继续处理右边的 ，这里是一个递归的过程
}

//写法2
void quickSortSort2(vector<int> &nums, int start, int end) {
    if (start >= end) {
        return;
    }

    int left = start, right = end;

    //选择锚点
    int pivot = nums[(start + end) / 2];

    while (left <= right)
    {
        //从左向右找，找到一个大于锚点的数
        while (left <= right && nums[left] < pivot)
        {
            left++;
        }

        //从右向左找，找到一个小于锚点的数
        while (left <= right && nums[right] > pivot)
        {
            right--;
        }

        //交换
        if (left <= right) {
            swap(nums[left],nums[right]);

            left++;
            right--;
        }
    }

    quickSortSort2(nums, start, right);
    quickSortSort2(nums, left, end);
}

Notes:

  1. 希尔排序gap的选择：通常选择增量gap = 数组length / 2，缩小增量继续以gap = gap / 2的方式。

  2. 快排的最坏情况的时间复杂度O（n^2）。
    原因：

    1. 在分解时每次选取的主元素为最小元素。
    2. 在分解时每次选取的主元素为最大元素。

    如何改善：三值取中法，即取左端、中间、右端三个数，然后进行排序，将中间数作为锚点值。
