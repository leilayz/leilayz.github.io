C++ 友元函数和友元类

2020-11-20 3 min read # C++

友元函数

目的：让非成员函数即普通函数或其他类可以访问类的私有成员。
优点：友元的作用在于提高程序的运行效率（即减少了类型检查和安全性检查等都需要的时间开销）。
缺点：破坏了类的封装性和隐藏性，使得非成员函数可以访问类的私有成员。

#include <iostream>
#include <string>
#include <iostream>
 
using namespace std;
 
class Box
{
   double width;
public:
   friend void printWidth( Box box );
   void setWidth( double wid );
};
 
// 成员函数定义
void Box::setWidth( double wid )
{
    width = wid;
}
 
// 请注意：printWidth() 不是任何类的成员函数
void printWidth( Box box )
{
   /* 因为 printWidth() 是 Box 的友元，它可以直接访问该类的任何成员 */
   cout << "Width of box : " << box.width <<endl;
}
 
// 程序的主函数
int main( )
{
   Box box;
 
   // 使用成员函数设置宽度
   box.setWidth(10.0);
   
   // 使用友元函数输出宽度
   printWidth( box );
 
   return 0;
}

成员函数类型的友元函数

将Volume类的一个成员函数在Box类中声明为友元函数，使得Volume类对象可访问Box类中所有成员。

#include <iostream>
#include <string>
#include <iostream>

using namespace std;

class Box;//类的前向声明
 
class Volume {
public:
   double getVolume(Box box);
};

class Box
{
   double width;
   double length;
public:
   friend double Volume::getVolume(Box box);
   void setWidth(double wid);
   void setLength(double len);
};
 
// 成员函数定义
void Box::setWidth(double wid)
{
    width = wid;
}

void Box::setLength(double len)
{
    length = len;
}

//注意成员函数类的友元函数的位置，要定义在类之后
double Volume::getVolume(Box box) {
   return box.width * box.length;
}
 
// 程序的主函数
int main( )
{
   Box box;
 
   // 使用成员函数设置宽度
   box.setWidth(10.0);
   box.setLength(20.0);

   Volume v;
   cout<<v.getVolume(box);
 
   return 0;
}

友元类

将Volume类定义为Box类的友元类后，Volume类可以访问Box类的所有成员。

#include <iostream>
#include <string>
#include <iostream>

using namespace std;

class Box;//类的前向声明
 
class Volume {
public:
   double getVolume(Box box);
};

class Box
{
   double width;
   double length;
public:
   friend class Volume;
   void setWidth(double wid);
   void setLength(double len);
};
 
// 成员函数定义
void Box::setWidth(double wid)
{
    width = wid;
}

void Box::setLength(double len)
{
    length = len;
}

//注意成员函数类的友元函数的位置，要定义在类之后
double Volume::getVolume(Box box) {
   return box.width * box.length;
}
 
// 程序的主函数
int main( )
{
   Box box;
 
   // 使用成员函数设置宽度
   box.setWidth(10.0);
   box.setLength(20.0);

   Volume v;
   cout<<v.getVolume(box);
 
   return 0;
}
